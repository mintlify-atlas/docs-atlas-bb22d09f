## Core concept: Word2Vec skip-gram from scratch

Word2Vec learns dense word representations by training a shallow neural network to predict context words from a center word. The learned weight matrix becomes the embedding lookup table.

```python
import numpy as np

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

class SkipGramWord2Vec:
    """Word2Vec skip-gram with negative sampling."""

    def __init__(self, vocab_size, embed_dim, lr=0.01):
        # Two embedding matrices: center words and context words
        self.W_in  = np.random.randn(vocab_size, embed_dim) * 0.01
        self.W_out = np.random.randn(vocab_size, embed_dim) * 0.01
        self.lr = lr

    def forward(self, center_idx, context_idx, negative_idxs):
        h = self.W_in[center_idx]                     # center word vector
        pos_score = sigmoid(h @ self.W_out[context_idx])   # positive pair
        neg_scores = sigmoid(-h @ self.W_out[negative_idxs].T)  # noise pairs

        # Binary cross-entropy loss
        loss = -np.log(pos_score + 1e-9) - np.sum(np.log(neg_scores + 1e-9))
        return loss, h, pos_score, neg_scores

    def get_embedding(self, word_idx):
        """Average of input and output embeddings at inference time."""
        return (self.W_in[word_idx] + self.W_out[word_idx]) / 2

# After training, find nearest neighbors with cosine similarity
def nearest_neighbors(query_vec, embeddings, top_k=5):
    norms = np.linalg.norm(embeddings, axis=1, keepdims=True) + 1e-9
    similarities = embeddings @ query_vec / (norms.squeeze() * np.linalg.norm(query_vec))
    return np.argsort(similarities)[::-1][:top_k]
```

The insight: words that appear in similar contexts end up with similar vectors, even without any explicit semantic labels. This is the foundation on which modern contextualized embeddings (BERT, GPT) are built.
