## Core concept: the mel spectrogram

Raw audio waveforms are hard to learn from directly. The mel spectrogram converts a waveform into a 2D time-frequency image that reflects how the human ear perceives pitch — making it the standard input feature for most speech and audio models.

```python
import numpy as np

def stft(signal, n_fft=512, hop_length=128, window=None):
    """Short-Time Fourier Transform — the building block of spectrograms."""
    if window is None:
        window = np.hanning(n_fft)

    n_frames = 1 + (len(signal) - n_fft) // hop_length
    spectrogram = np.zeros((n_fft // 2 + 1, n_frames), dtype=complex)

    for t in range(n_frames):
        start = t * hop_length
        frame = signal[start:start + n_fft] * window
        spectrogram[:, t] = np.fft.rfft(frame)

    return spectrogram

def mel_filterbank(n_filters, n_fft, sr, f_min=0, f_max=None):
    """Build a mel filterbank matrix: (n_filters, n_fft // 2 + 1)."""
    if f_max is None:
        f_max = sr / 2

    # Convert Hz to mel
    mel_min = 2595 * np.log10(1 + f_min / 700)
    mel_max = 2595 * np.log10(1 + f_max / 700)
    mel_points = np.linspace(mel_min, mel_max, n_filters + 2)
    hz_points = 700 * (10 ** (mel_points / 2595) - 1)

    bins = np.floor((n_fft + 1) * hz_points / sr).astype(int)
    fbank = np.zeros((n_filters, n_fft // 2 + 1))

    for m in range(1, n_filters + 1):
        f_left, f_center, f_right = bins[m - 1], bins[m], bins[m + 1]
        for k in range(f_left, f_center):
            fbank[m - 1, k] = (k - f_left) / (f_center - f_left)
        for k in range(f_center, f_right):
            fbank[m - 1, k] = (f_right - k) / (f_right - f_center)

    return fbank

def mel_spectrogram(signal, sr=16000, n_fft=512, hop_length=128, n_mels=80):
    S = np.abs(stft(signal, n_fft, hop_length)) ** 2
    mel_fb = mel_filterbank(n_mels, n_fft, sr)
    mel_S = mel_fb @ S
    return np.log(mel_S + 1e-9)  # log scale; shape: (n_mels, n_frames)
```

Whisper takes exactly this representation — an 80-channel log-mel spectrogram — as its encoder input, treating it as a 2D image processed through convolutional layers before the transformer stack.
