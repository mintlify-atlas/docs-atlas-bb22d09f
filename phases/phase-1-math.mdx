Every AI algorithm — from a simple linear classifier to a billion-parameter language model — reduces to a handful of mathematical operations: matrix multiplications, gradient computations, probability distributions, and optimization steps. Phase 1 builds genuine intuition for each of these, not through abstract proofs, but through code. Every concept is implemented from scratch in Python, with several lessons offering parallel Julia implementations to show how numerical computing looks in a language designed for it. By the end, you won't just recognize terms like "eigenvalue" or "KL divergence" — you'll have built them and seen exactly how they behave.

```python
# Lesson 05: Chain Rule & Autodiff — scalar autodiff engine
class Value:
    """A scalar value with a gradient tape for automatic differentiation."""
    def __init__(self, data, _children=(), _op=''):
        self.data = data
        self.grad = 0.0
        self._backward = lambda: None
        self._prev = set(_children)
        self._op = _op

    def __mul__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out = Value(self.data * other.data, (self, other), '*')

        def _backward():
            self.grad  += other.data * out.grad
            other.grad += self.data  * out.grad
        out._backward = _backward
        return out

    def __add__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out = Value(self.data + other.data, (self, other), '+')

        def _backward():
            self.grad  += out.grad
            other.grad += out.grad
        out._backward = _backward
        return out

    def backward(self):
        # Topological sort then reverse-mode pass
        topo, visited = [], set()
        def build(v):
            if v not in visited:
                visited.add(v)
                for child in v._prev:
                    build(child)
                topo.append(v)
        build(self)
        self.grad = 1.0
        for node in reversed(topo):
            node._backward()

# Usage
x = Value(2.0)
y = Value(3.0)
z = x * y + x   # z = 2*3 + 2 = 8
z.backward()
print(x.grad)   # dz/dx = y + 1 = 4.0
print(y.grad)   # dz/dy = x     = 2.0
```
